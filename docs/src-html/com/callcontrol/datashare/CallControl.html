135    /**
136     * Reported number. Contain a number and possibly a name to go along with it.
137     */
138    public static final class Report implements Parcelable {
139        /**
140         * Number being reported.
141         */
142        private String number;
143        /**
144         * Known name for the number.
145         * <p>Optional.
146         */
147        private String name;
148        /**
149         * Whether or not the number is unwanted.
150         */
151        private boolean isUnwanted;
152
153        /**
154         * Constructs the object assuming the number is unwanted.
155         *
156         * @param number Number being reported.
157         */
158        public Report(@NonNull String number) {
159            this(number, null, true);
160        }
161
162        /**
163         * Constructs the object assuming the number is unwanted.
164         *
165         * @param number Number being reported.
166         * @param name   Name known for the number.
167         */
168        public Report(@NonNull String number, @Nullable String name) {
169            this(number, name, true);
170        }
171
172        /**
173         * Constructs the object.
174         *
175         * @param number     Number being reported.
176         * @param name       Name known for the number.
177         * @param isUnwanted Indicates if user doesn't want to be contacted by the number.
178         */
179        public Report(@NonNull String number, @Nullable String name, boolean isUnwanted) {
180            this.number = number;
181            this.name = name;
182            this.isUnwanted = isUnwanted;
183        }
184
185        /**
186         * Constructs the object form parcel.
187         *
188         * @param parcel Parcel.
189         */
190        Report(Parcel parcel) {
191            number = parcel.readString();
192            name = parcel.readString();
193            isUnwanted = parcel.readInt() == 1;
194        }
195
196        /**
197         * Returns the associated phone number.
198         *
199         * @return Associated phone number.
200         */
201        @NonNull
202        public String getNumber() {
203            return number;
204        }
205
206        /**
207         * Returns the associated name, if any.
208         *
209         * @return Associated name, if any.
210         */
211        @Nullable
212        public String getName() {
213            return name;
214        }
215
216        /**
217         * Returns whether or not user refuses to receive any activity from the phone number.
218         *
219         * @return Whether or not user refuses to receive any activity from the phone number.
220         */
221        public boolean isUnwanted() {
222            return isUnwanted;
223        }
224
225        @Override
226        public void writeToParcel(Parcel dest, int flags) {
227            dest.writeString(number);
228            dest.writeString(name);
229            dest.writeInt(isUnwanted ? 1 : 0);
230        }
231
232        @Override
233        public int describeContents() {
234            return 0;
235        }
236
237        /**
238         * Parcel interface used to create instances from parcels.
239         */
240        public static final Parcelable.Creator<Report> CREATOR = new Parcelable.Creator<Report>() {
241            @Override
242            public Report createFromParcel(Parcel source) {
243                return new Report(source);
244            }
245
246            @Override
247            public Report[] newArray(int size) {
248                return new Report[size];
249            }
250        };
251    }

263    /**
264     * Opens specified number in Blocked List in Call Control if possible.
265     * <p>If the number is null, the behavior is identical to {@link #openBlockedList(Context)}.
266     *
267     * @param context     Context where {@link Context#startActivity(Intent)} will be executed.
268     * @param phoneNumber Phone number to present.
269     * @return True if the intent has been sent, false on error.
270     */
271    public static boolean openBlockedList(@NonNull Context context, @Nullable String phoneNumber) {
272        Intent i = new Intent(ACTION_BLOCKED_LIST);
273        if (null != phoneNumber) i.putExtra(EXTRA_PHONE_NUMBER, phoneNumber);
274        if (!signIntent(context, i)) return false;
275        context.startActivity(i);
276        return true;
277    }

309    /**
310     * <p>Reports a set of numbers via Call Control UI.
311     * <p>Will present Call Control UI which interacts with user.
312     * <p>After user confirms the report, the number is also added to Blocked List.
313     * <p>
314     * <p><b>IMPORTANT:</b> Must contain the list with ALL items having the same value of the {@link Report#isUnwanted} flag.
315     * I.e. in one run you only report either unwanted or wanted numbers, but you cannot mix them in one call.
316     * <p>
317     * <p>Safe to execute with empty list, does nothing in this case.
318     * <p>
319     * <p>Useful when user enables the integration and the application wants to put already known
320     * blocked numbers to Call Control in one run.
321     *
322     * @throws IllegalArgumentException When {@link Report#isUnwanted} is not the same for all numbers.
323     *
324     * @param context    Context where {@link Context#startActivity(Intent)} will be executed.
325     * @param reports    List of numbers to report. Must all have identical {@link Report#isUnwanted} flag.
326     * @return True if the intent has been sent, false on error.
327     */
328    public static boolean report(@NonNull Context context, @NonNull ArrayList<Report> reports) {
329        if (reports.isEmpty()) return false;
330        boolean isUnwanted = reports.get(0).isUnwanted;
331        for (Report n : reports) {
332            if (n.isUnwanted != isUnwanted) throw new IllegalArgumentException("All numbers reported must have identical isUnwanted flag");
333        }
334        Intent i = new Intent(ACTION_REPORT);
335        i.putExtra(EXTRA_PHONE_NUMBER, reports);
336        if (!signIntent(context, i)) return false;
337        context.startActivity(i);
338        return true;
339    }

355    /**
356     * Performs intent "signing", i.e. requesting and adding {@link #EXTRA_TOKEN} extra to the intent.
357     * <p>Intents that do not contain a valid {@link #EXTRA_TOKEN} will not be honored by Call Control.
358     *
359     * @param context Context where the {@link ContentResolver#query(Uri, String[], String, String[], String)}
360     *                will be executed.
361     * @param intent  Intent to "sign".
362     * @return True if the sign succeeded, false otherwise.
363     */
364    private static boolean signIntent(@NonNull Context context, @NonNull Intent intent) {
365        try {
366            long currentTime = System.currentTimeMillis();
367            String sender = context.getPackageName();
368            ContentResolver resolver = context.getContentResolver();
369            Cursor c = resolver.query(REQUEST_TOKEN_URI, null, Token.TIMESTAMP+" = ?", new String[]{ String.valueOf(currentTime) }, null);
370            if (null == c) return false;
371            String token = "";
372            if (c.moveToFirst()) token = c.getString(c.getColumnIndex(Token.TOKEN));
373            c.close();
374
375            ByteBuffer buf = ByteBuffer.allocate((Long.SIZE / Byte.SIZE) + token.getBytes().length + sender.getBytes().length);
376            buf.putLong(currentTime);
377            buf.put(token.getBytes());
378            buf.put(sender.getBytes());
379            intent.putExtra(EXTRA_TOKEN, new String(Base64.encode(buf.array(), Base64.DEFAULT)));
380            return true;
381        } catch (Throwable ignore) {}
382        return false;
383    }
